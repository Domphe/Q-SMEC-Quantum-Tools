### How To: Query Database

```python
import sqlite3
import pandas as pd

# Connect to database
DB_PATH = r"G:\My Drive\Databases\Quantumn AI DB _ Tools\quantum_ai_tools.db"
conn = sqlite3.connect(DB_PATH)

# Query 1: Top Q-SMEC files
query = """
SELECT file_name, qsmec_relevance_score, ai_relevance_score, quantum_relevance_score
FROM ai_tools_files
WHERE qsmec_relevance_score > 20
ORDER BY qsmec_relevance_score DESC
LIMIT 10
"""
df = pd.read_sql_query(query, conn)
print(df)

# Query 2: AI + Quantum integration pairs
query_pairs = """
SELECT af.file_name, ai.framework_name, qc.capability_name
FROM ai_tools_files af
JOIN ai_frameworks ai ON af.id = ai.file_id
JOIN quantum_capabilities qc ON af.id = qc.file_id
WHERE af.ai_relevance_score > 30 AND af.quantum_relevance_score > 15
LIMIT 20
"""
df_pairs = pd.read_sql_query(query_pairs, conn)
print(df_pairs)

conn.close()
```
